char S[100010], T[100010], str[260];

int main()
{
    memset(str, 0, sizeof(0));
    char ch = 'a', cc;
    int count = 0;
    while (cc = getchar())
    {
        if (cc >= 'a' && cc <= 'z')
            S[count++] = cc;
        else
            break;
    }
    S[count] = '\n';
    for (int i=0;i<count;++i)
    {
        if (str[S[i]] == 0)
        {
            str[S[i]] = ch;
            T[i] = ch;
            ch++;
        } else {
            T[i] = str[S[i]];
        }
    }
    T[count] = '\0';
    for (int i=0;i<count;++i)
        putchar(T[i]);
    return 0;
}

function lcs(x, y)
{
	for (var i=0;i<Math.min(x.length, y.length);++i)
    	if (x.substring(x.length - i - 1) == y.substring(0, i + 1))
              return y.substring(0, i + 1);
    return '';
}
